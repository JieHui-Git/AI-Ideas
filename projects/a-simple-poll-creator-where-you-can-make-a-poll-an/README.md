## Setup

1. cd backend && pip install -r requirements.txt
2. npm init -y
3. echo 'frontend/index.html' > gitignore
4. echo 'node_modules' >> gitignore
5. echo '.env' >> gitignore
6. echo '/dist' >> gitignore
7. cd frontend && npx create-vite@latest .
8. cd ..
9. npm install express sqlite3 cors body-parser dotenv nodemon
10. Run backend: nodemon -r dotenv/config main.py --watch main.py
11. Run frontend: cd frontend && npm run dev
